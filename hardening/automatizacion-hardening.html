Playbook B√°sico de Hardening

# playbooks/harden-linux.yml
---
- name: Hardening b√°sico de Linux
  hosts: all
  become: yes
  tasks:
    
    - name: Actualizar sistema
      apt:
        update_cache: yes
        upgrade: dist
      when: ansible_os_family == "Debian"
    
    - name: Instalar paquetes de seguridad
      apt:
        name:
          - ufw
          - fail2ban
          - unattended-upgrades
        state: present
      when: ansible_os_family == "Debian"
    
    - name: Configurar UFW
      ufw:
        rule: allow
        port: "{{ item }}"
        proto: tcp
      loop:
        - 22
        - 80
        - 443
      notify: restart ufw
    
    - name: Habilitar UFW
      ufw:
        state: enabled
    
    - name: Deshabilitar root login SSH
      lineinfile:
        path: /etc/ssh/sshd_config
        regexp: '^PermitRootLogin'
        line: 'PermitRootLogin no'
        backup: yes
      notify: restart sshd
    
    - name: Configurar Fail2ban
      copy:
        src: fail2ban.local
        dest: /etc/fail2ban/jail.local
      notify: restart fail2ban
    
  handlers:
    - name: restart sshd
      systemd:
        name: sshd
        state: restarted
    
    - name: restart ufw
      systemd:
        name: ufw
        state: restarted
    
    - name: restart fail2ban
      systemd:
        name: fail2ban
        state: restarted

# Ejecutar playbook
ansible-playbook -i inventory/hosts playbooks/harden-linux.yml

Usar Roles de Ansible

# O usar role de CIS Benchmarks

# Crear playbook que usa el role

---
- name: Hardening con role
  hosts: all
  become: yes
  roles:
    - devsec.hardening
  vars:
    os_hardening_ssh_permit_root_login: false
    os_hardening_ssh_password_authentication: false
EOF

# Ejecutar
ansible-playbook -i inventory/hosts playbooks/harden-with-role.yml

# Crear role personalizado

Scripts Bash para Hardening

#!/bin/bash
# Script de hardening b√°sico
# Guardar como: harden-server.sh

set -euo pipefail

# Colores para output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

log_info() {
    echo -e "${GREEN}[INFO]${NC} $1"
}

log_warn() {
    echo -e "${YELLOW}[WARN]${NC} $1"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

# Verificar que se ejecuta como root
if [[ $EUID -ne 0 ]]; then
   log_error "Este script debe ejecutarse como root"
   exit 1
fi

log_info "Iniciando hardening del servidor..."

# 1. Actualizar sistema
log_info "Actualizando sistema..."
apt update && apt upgrade -y

# 2. Instalar herramientas de seguridad
log_info "Instalando herramientas de seguridad..."
apt install -y ufw fail2ban unattended-upgrades apt-listchanges

# 3. Configurar UFW
log_info "Configurando firewall..."
ufw default deny incoming
ufw default allow outgoing
ufw allow 22/tcp
ufw allow 80/tcp
ufw allow 443/tcp
ufw --force enable

# 4. Configurar SSH
log_info "Configurando SSH..."
cp /etc/ssh/sshd_config /etc/ssh/sshd_config.backup
sed -i 's/#PermitRootLogin yes/PermitRootLogin no/' /etc/ssh/sshd_config
sed -i 's/#PasswordAuthentication yes/PasswordAuthentication no/' /etc/ssh/sshd_config
systemctl restart sshd

# 5. Configurar Fail2ban
log_info "Configurando Fail2ban..."
cp /etc/fail2ban/jail.conf /etc/fail2ban/jail.local
systemctl enable fail2ban
systemctl start fail2ban

# 6. Habilitar actualizaciones autom√°ticas
log_info "Configurando actualizaciones autom√°ticas..."
echo 'Unattended-Upgrade::Automatic-Reboot "false";' >> /etc/apt/apt.conf.d/50unattended-upgrades

log_info "Hardening completado exitosamente!"
log_warn "Recuerda verificar que puedes conectarte por SSH antes de cerrar esta sesi√≥n!"
