Automatizaci√≥n

Puedes integrar OpenSCAP en scripts o CI/CD:

# Script de evaluaci√≥n automatizada

SCAP_FILE="/usr/share/xml/scap/ssg/content/ssg-ubuntu2204-ds.xml"
PROFILE="xccdf_org.ssgproject.content_profile_cis"
RESULTS_DIR="/var/log/openscap"
DATE=$(date +%Y%m%d_%H%M%S)

mkdir -p $RESULTS_DIR

sudo oscap xccdf eval \
  --profile $PROFILE \
  --results $RESULTS_DIR/results-$DATE.xml \
  --report $RESULTS_DIR/report-$DATE.html \
  $SCAP_FILE

# Extraer score
SCORE=$(oscap xccdf eval --profile $PROFILE $SCAP_FILE 2>&1 | \
  grep -oP 'Score: \K[0-9]+')

echo "Compliance Score: $SCORE"

# Enviar alerta si score es bajo
if [ "$SCORE" -lt 70 ]; then
    echo "ALERTA: Score de compliance bajo ($SCORE)" | \
      mail -s "OpenSCAP Report - Low Score" admin@example.com
fi
