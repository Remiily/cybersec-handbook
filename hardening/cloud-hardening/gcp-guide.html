IAM y Permisos

# Listar roles disponibles
gcloud iam roles list

# Crear custom role
gcloud iam roles create myCustomRole --project=PROJECT_ID --title="Custom Role" --description="Role personalizado" --permissions=compute.instances.list,compute.instances.get

# Asignar rol a usuario
gcloud projects add-iam-policy-binding PROJECT_ID --member="user:usuario@dominio.com" --role="roles/compute.instanceAdmin"

# Asignar rol a service account
gcloud projects add-iam-policy-binding PROJECT_ID --member="serviceAccount:sa@project.iam.gserviceaccount.com" --role="roles/storage.objectViewer"

# Listar pol√≠ticas IAM del proyecto

Firewall Rules

# Crear regla de firewall para permitir HTTP
gcloud compute firewall-rules create allow-http --allow tcp:80 --source-ranges 0.0.0.0/0 --description "Allow HTTP traffic"

# Crear regla para permitir SSH solo desde IP espec√≠fica
gcloud compute firewall-rules create allow-ssh-from-office --allow tcp:22 --source-ranges 203.0.113.0/24 --description "Allow SSH from office"

# Crear regla de firewall con tags
gcloud compute firewall-rules create allow-internal --allow tcp:8080 --source-tags web-server --target-tags app-server

# Listar reglas de firewall
gcloud compute firewall-rules list

# Eliminar regla
gcloud compute firewall-rules delete allow-http
