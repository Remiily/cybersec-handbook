Integraci√≥n en CI/CD

Los SBOMs deben generarse autom√°ticamente en cada build para mantener un registro completo de dependencias.

GitHub Actions

# .github/workflows/sbom.yml
name: Generate SBOM

on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]

jobs:
  generate-sbom:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v3
    
    - name: Install Syft
      run: |
        curl -sSfL https://raw.githubusercontent.com/anchore/syft/main/install.sh | sh -s -- -b ${{ runner.temp }}
        echo "${{ runner.temp }}" >> $GITHUB_PATH
    
    - name: Generate SBOM
      run: |
        syft dir:. -o cyclonedx-json > sbom.json
    
    - name: Upload SBOM artifact
      uses: actions/upload-artifact@v3
      with:
        name: sbom
        path: sbom.json
    
    - name: Scan SBOM for vulnerabilities
      run: |
        curl -sSfL https://raw.githubusercontent.com/anchore/grype/main/install.sh | sh -s -- -b ${{ runner.temp }}
        grype sbom:./sbom.json --fail-on high
    
    - name: Upload SBOM to GitHub
      uses: github/codeql-action/upload-sarif@v2
      if: always()
      with:
        sarif_file: vulnerabilities.json
