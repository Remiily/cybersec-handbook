Generar SBOM

Syft (Anchore)

# Instalar Syft
# macOS
brew install syft

# Linux
curl -sSfL https://raw.githubusercontent.com/anchore/syft/main/install.sh | sh -s -- -b /usr/local/bin

# Generar SBOM de imagen Docker
syft docker-image nginx:latest -o cyclonedx-json > sbom.json

# Generar SBOM de directorio
syft dir:/ruta/proyecto -o cyclonedx-json > sbom.json

# Generar SBOM de paquete npm
syft packages npm:/ruta/proyecto -o cyclonedx-json > sbom.json

# Formatos disponibles: cyclonedx-json, cyclonedx-xml, spdx-json, spdx-tag-value

SPDX Tools

# Instalar SPDX tools
pip install spdx-tools

# Generar SBOM SPDX
spdx-tools generate -i /ruta/proyecto -o sbom.spdx

# Validar SBOM
spdx-tools validate sbom.spdx

# Convertir entre formatos
spdx-tools convert -i sbom.spdx -o sbom.json --format json

Analizar SBOM para Vulnerabilidades

# Grype - Analizar SBOM
# Instalar
brew install grype
# O
curl -sSfL https://raw.githubusercontent.com/anchore/grype/main/install.sh | sh -s -- -b /usr/local/bin

# Analizar imagen directamente
grype docker:nginx:latest

# Analizar SBOM
grype sbom:./sbom.json

# Output JSON
grype docker:nginx:latest -o json > vulnerabilities.json

# Filtrar por severidad
grype docker:nginx:latest --only-fixed --fail-on high
